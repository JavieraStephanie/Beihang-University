int factorial(int n)
{   
    if(n==1)    return (1);
    return (n*factorial(n-1));
}
void main()
{
    int num;
    int fact;
     
    fact=0;
    printf("Enter an integer number: ");
    scanf(num);
     
    fact=factorial(num);
    printf("Factorial of %d is = %ld");
    printf(num); 
    printf(fact); 
    printf("\n");
}
