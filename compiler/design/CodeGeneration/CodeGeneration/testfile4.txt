void main()
{
    int count, first_term, second_term, next_term, i;
 
    first_term = 0;
    second_term = 1;
    
    printf("Enter the number of terms:\n");
    scanf(count);
 
    printf("First terms of Fibonacci series:\n",count);
    for ( i = 0 ; i < count ; i = i + 1 )
    {
       if ( i <= 1 )
          next_term = i;
       else
       {
          next_term = first_term + second_term;
          first_term = second_term;
          second_term = next_term;
       }
       printf(next_term);
    }
}
